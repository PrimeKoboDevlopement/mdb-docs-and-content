import { MDBFilters } from 'mdb-react-filters';

Properties

Filters

Name Type Default Description Example
className string '' Add custom class to MDBFilters. <MDBFilters className="class" />
data Array: any - Array with the original data. <MDBFilters data={exampleData} />
filterData React.SetStateAction - Intented to set the state of the filtered data. <MDBFilters filterData={setData} />
filteredData any - State with the filtered data. <MDBFilters filteredData={exampleData} />
filterOptions any - Defines the filter options. <MDBFilters filterOptions={options} />
sortOptions any - Defines the sort options. <MDBFilters sortOptions={options} />
